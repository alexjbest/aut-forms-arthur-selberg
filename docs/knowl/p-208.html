\begin{equation*} \tilde \phi_0(s) = \int_0^1 \sum_{n\in \ZZ} a_n \overline b_n e^{-4 \pi i y} y^{k+s}\frac{\diff y}{y} \end{equation*}
\begin{equation*} = \sum_{n\in \ZZ} a_n\overline b_n\int_0^\infty e^{-4 \pi i y} y^{k+s}\frac{\diff y}{y} \end{equation*}
\begin{equation*} = \frac{1}{(4\pi )^s} \sum_{n\in \ZZ}\frac{ a_n\overline b_n}{n^{k+s}}\int_0^\infty e^{-y} y^{k+s}\frac{\diff y}{y} \end{equation*}
\begin{equation*} = \frac{\Gamma(k+s)}{(4\pi)^{k+s}}\underbrace{\sum_{n=1}^\infty\frac{ a_n\overline b_n}{n^{k+s}}}_{L(s+k; f\times \overline g)} \end{equation*}
